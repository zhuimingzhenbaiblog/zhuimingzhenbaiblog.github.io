c++新添知识1.0

继承：
class BasePage
{
public:
……
};
class java : public BasePage
{
public:
…
};
语法：
class 子类 : 继承方式 父类 {};
继承方式：
1.公共继承
2.保护继承
3.私有继承

父类中的私有内容不可以访问，3种都不行
在父类中的所有非静态成员属性都会被子类继承下去，私有成员被继承，但是不能访问
如果想访问父类中的被隐藏的同名成员或者成员函数，需要加作用域::

静态成员属性访问方式：
1.通过对象访问：
s.Base::m_A
2.通过类名访问:
son::m_A / Base::m_A

c++中允许一个类继承多个类
语法：
class 子类 ： 继承方式 父类1 ， 继承方式 父类2 {};

关于菱形继承：
需要虚继承： 在继承之前 加上关键字 virtual 变成虚继承
class Sheep : virtual public Animal {}; //Animal类称为 虚基类

虚函数： virtual 返回类型 函数名 (){}
纯虚函数： virtual 返回类型 函数名 () = 0;
动态多态满足关系：
1.有继承关系
2.子类重写父类的虚函数
动态多态的使用：
父类的指针或者引用 执行子类对象

纯虚函数：
只要有一个纯虚函数，这个类就称为抽象类
特点：
1.无法实例化对象
2.抽象类的子类，必须要重写父类中的纯虚数函数，否则也属于抽象类

多态案例：
#include
using namespace std;

class AbstractDrinking //制作饮品抽象类
{
public:
virtual void Boil() = 0; //煮水
virtual void Brew() = 0; //冲泡
virtual void PourInCup() = 0; //倒入杯中
virtual void PutSomething() = 0; //加入辅料
void makeDrink()
{
Boil();
Boil();
PourInCup();
PutSomething();
}
};
class Coffee : public AbstractDrinking
{
public:
virtual void Boil() //煮水
{
cout << “煮水” << endl;
}
virtual void Brew() //冲泡
{
cout << “冲泡咖啡” << endl;
}
virtual void PourInCup() //倒入杯中
{
cout << “倒入杯中” << endl;
}
virtual void PutSomething() //加入辅料
{
cout << “加入糖和牛奶” << endl;
}
};
class Tea : public AbstractDrinking
{
public:
virtual void Boil() //煮水
{
cout << “煮茶水” << endl;
}
virtual void Brew() //冲泡
{
cout << “冲泡茶叶” << endl;
}
virtual void PourInCup() //倒入杯中
{
cout << “倒入茶杯中” << endl;
}
virtual void PutSomething() //加入辅料
{
cout << “加入柠檬” << endl;
}
};
void doWork(AbstractDrinking* abs) //制作函数
{
abs->makeDrink();
delete abs; //释放空间
}
void test() //制作咖啡
{
doWork(new Coffee); //AbstractDrinking* abs = new Coffee
}
void test1() //制作茶
{
doWork(new Tea); //AbstractDrinking* abs = new Tea
}
void doWork(AbstractDrinking& abs) //制作函数 用引用方法
{
abs.makeDrink();
}
void test2()
{
Tea tea;
doWork(tea);
}

int main() {

test();
test1();
test2();
system("pause");
return 0;

}

虚析构：
virtual ~Animal(){} //利用虚析构可以解决父类指针释放子类对象时不干净的问题
要点：
1.虚析构或纯虚析构是用来解决父类指针释放子类对象
2.如果子类中没有堆区的数据，可以不写虚析构或者纯虚析构
3.拥有纯虚析构函数的类也属于抽象类

案例2：
#include
using namespace std;

class CPU //抽象CPU类
{
public:
virtual void calculate() = 0; //抽象计算函数
};

class GPU //抽象GPU类
{
public:
virtual void display() = 0; //抽象显示函数
};

class RAM //抽象内存类
{
public:
virtual void storage() = 0; //抽象存储函数
};

class Computer //电脑类
{
public:
Computer(CPU* cpu, GPU* gpu, RAM* ram) //有参构造函数
{
m_cpu = cpu;
m_gpu = gpu;
m_ram = ram;
}
void work() //工作函数
{
m_cpu->calculate();
m_gpu->display();
m_ram->storage();
}
//提供析构函数 释放3个电脑零件
~Computer()
{
if (m_cpu != NULL)
{
delete m_cpu;
m_cpu = NULL;
}
if (m_gpu != NULL)
{
delete m_gpu;
m_gpu = NULL;
}
if (m_ram!= NULL)
{
delete m_ram;
m_ram = NULL;
}
}
private:
CPU* m_cpu; //CPU的零件指针
GPU* m_gpu; //GPU的零件指针
RAM* m_ram; //RAM的零件指针
};
//具体厂商Intel
class IntelCPU : public CPU
{
public:
void calculate()
{
cout << “Intel的CPU开始工作” << endl;
}
};
class IntelGPU : public GPU
{
public:
void display()
{
cout << “Intel的GPU开始工作” << endl;
}
};
class IntelRAM : public RAM
{
public:
void storage()
{
cout << “Intel的内存条开始工作” << endl;
}
};
//具体厂商AMD
class AMDCPU : public CPU
{
public:
void calculate()
{
cout << “AMD的CPU开始工作” << endl;
}
};
class AMDGPU : public GPU
{
public:
void display()
{
cout << “AMD的GPU开始工作” << endl;
}
};
class AMDRAM : public RAM
{
public:
void storage()
{
cout << “AMD的内存条开始工作” << endl;
}
};

void test() //第一台机器的零件
{
CPU* intelcpu = new IntelCPU;
GPU* intelgpu = new IntelGPU;
RAM* intelram = new IntelRAM;
Computer* computer1 = new Computer(intelcpu, intelgpu, intelram);
computer1->work();
delete computer1;

}
int main() {
test();
system(“pause”);
return 0;
}
